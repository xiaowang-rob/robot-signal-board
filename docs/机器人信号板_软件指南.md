# 机器人信号板 — 软件开发指南

> 基于 HPM6E80IVM1 的多总线通信汇聚板
> 双核 RISC-V @ 600MHz，2080KB SRAM

## 1. 开发环境搭建

### 1.1 所需工具链

| 工具 | 说明 | 下载 |

| **HPM SDK** | 先楫官方 SDK，含驱动和中间件 | https://github.com/hpmicro/hpm_sdk |
| **RISC-V GCC** | 交叉编译工具链 | SDK 内置或官方下载 |

| **VS Code** | 推荐 IDE | 可选 |

```bash
# 克隆 SDK
```

### 2.2 双核任务分配

| 核心 | 职责 | 任务 |
|------|------|------|
| **CPU0** (主核) | 数据汇聚 + 通信 | USB 发送、以太网服务、CAN 接收、状态监测 |
| **CPU1** (辅助核) | 总线数据采集 | RS485 接收、业务串口、HLK-7628N 管理、预处理 |

**核间通信**: 通过邮箱 MBX + 共享内存 (AXI SRAM)

```
    CPU0                     CPU1
    ┌─────────┐             ┌─────────┐
    │ USB 发送│             │ RS485 收│
    │ 以太网  │ ←─ MBX ──→ │ 串口管理│
    │ CAN 汇聚│  共享内存   │ 预处理  │
    │ 状态监测│             │ HLK管理 │
    └─────────┘             └─────────┘
```

### 3.2 UART 驱动

```c
#include "hpm_uart_drv.h"

// UART 配置结构
typedef struct {
    UART_Type *base;        // UART 基地址 (HPM_UART0~HPM_UART16)
    uint32_t baudrate;      // 波特率
    uint8_t tx_pin;         // TX 引脚号
    uint8_t rx_pin;         // RX 引脚号
    uint8_t de_pin;         // RS485 DE 引脚 (可选)
} uart_config_t;

// 初始化 UART (以 UART0, 115200bps 为例)
void uart_init(void)
{
    uart_config_t config = {
        .base = HPM_UART0,
        .baudrate = 115200,
    };

    uart_control_config_t ctrl_cfg = {
        .tx_enable = true,
        .rx_enable = true,
        .word_size = word_size_8_bits,
        .stop_bits = stop_bits_1,
        .parity = parity_none,
    };

    uart_init_control(UART0, &ctrl_cfg);
    uart_set_baudrate(UART0, clock_get_frequency(clock_uart0), config.baudrate);
}

// UART DMA 收发 (用于 RS485 高速传输)
void uart_dma_rx_start(UART_Type *base, uint8_t *buf, uint32_t len)
{
    // 配置 DMA 通道
    dma_channel_config_t dma_cfg = {
        .src_addr = (uint32_t)&base->RBR,
        .dst_addr = (uint32_t)buf,
        .size = len,
        .src_width = DMA_TRANSFER_WIDTH_BYTE,
        .dst_width = DMA_TRANSFER_WIDTH_BYTE,
        .src_mode = DMA_HANDSHAKE_MODE_NORMAL,
        .dst_mode = DMA_HANDSHAKE_MODE_NORMAL,
        .src_burst_size = DMA_NUM_TRANSFER_PER_BURST_1T,
    };

    dma_setup_channel(DMA, DMA_CH_UART0_RX, &dma_cfg, true);
    dma_enable_channel(DMA, DMA_CH_UART0_RX);
}
```

### 3.3 CAN-FD 驱动 (8 路原生 ⭐)

```c
#include "hpm_mcan_drv.h"

// 每路 CAN 配置
typedef struct {
    MCAN_Type *base;        // MCAN0~MCAN7
    uint8_t tx_pin;
    uint8_t rx_pin;
    uint8_t stby_pin;
    uint32_t bitrate;       // 标准速率 (kbps)
    uint32_t data_bitrate;  // 数据段速率 (kbps) - CAN-FD
} mcan_config_t;

// CAN 消息结构 (上位机协议兼容)
typedef struct __attribute__((packed)) {
    uint8_t channel;        // 通道号 0~7
    uint32_t timestamp;     // μs 时间戳
    uint32_t id;            // CAN ID (标准/扩展)
    uint8_t id_type;        // 0=标准, 1=扩展
    uint8_t frame_type;     // 0=数据帧, 1=远程帧
    uint8_t dlc;            // 数据长度
    uint8_t data[64];       // CAN-FD 最大 64 字节
    uint8_t flags;          // 错误标志位
} can_message_t;

// 初始化单路 CAN-FD
void mcan_init_single(MCAN_Type *base, uint32_t bitrate, uint32_t data_bitrate)
{
    mcan_config_t cfg;
    mcan_get_default_config(&cfg);

    cfg.baudrate = bitrate * 1000;       // 仲裁段速率
    cfg.data_baudrate = data_bitrate * 1000; // 数据段速率
    cfg.mode = mcan_mode_normal;
    cfg.enable_canfd = true;
    cfg.enable_ttcm = false;

    mcan_init(base, &cfg);
    mcan_enter_normal_mode(base);
}

// 初始化全部 8 路 CAN-FD
void mcan_init_all(void)
{
    // 可根据实际应用配置不同波特率
    mcan_init_single(HPM_MCAN0, 500, 2000);   // 500kbps/2Mbps
    mcan_init_single(HPM_MCAN1, 500, 2000);
    mcan_init_single(HPM_MCAN2, 500, 2000);
    mcan_init_single(HPM_MCAN3, 500, 2000);
    mcan_init_single(HPM_MCAN4, 1000, 4000);  // 1Mbps/4Mbps
    mcan_init_single(HPM_MCAN5, 1000, 4000);
    mcan_init_single(HPM_MCAN6, 1000, 4000);
    mcan_init_single(HPM_MCAN7, 1000, 4000);
}

// CAN 接收中断服务
void isr_mcan0(void)
{
    mcan_rx_message_t rx_msg;
    mcan_read_rx_fifo(HPM_MCAN0, &rx_msg);

    can_message_t msg = {
        .channel = 0,
        .timestamp = timer_get_us(),
        .id = rx_msg.id,
        .dlc = rx_msg.dlc,
    };
    memcpy(msg.data, rx_msg.data, rx_msg.dlc);

    // 入队到统一 FIFO
    xQueueSend(can_rx_queue, &msg, 0);
}

// CAN 发送
bool mcan_send(MCAN_Type *base, can_message_t *msg)
{
    mcan_tx_message_t tx_msg = {
        .id = msg->id,
        .dlc = msg->dlc,
        .tx_fifo = mcan_tx_fifo_0,
    };
    memcpy(tx_msg.data, msg->data, msg->dlc);

    return (mcan_send_message(base, &tx_msg) == status_success);
}
```

## 4. 中间件配置

### 4.1 FreeRTOS 任务定义

```c
// 任务优先级
#define PRIO_CAN_RX      (configMAX_PRIORITIES - 1)  // 最高: CAN 实时
#define PRIO_RS485_RX    (configMAX_PRIORITIES - 2)
#define PRIO_ETH_SERVICE (configMAX_PRIORITIES - 3)
#define PRIO_USB_TX      (configMAX_PRIORITIES - 3)
#define PRIO_MONITOR     (configMAX_PRIORITIES - 4)

// 任务堆栈大小
#define STACK_CAN_RX     4096
#define STACK_RS485_RX   2048
#define STACK_USB_TX     4096
#define STACK_MONITOR    1024

// 创建任务
void app_task_init(void)
{
    // CAN 接收 (CPU0)
    xTaskCreate(can_rx_task, "CAN_RX", STACK_CAN_RX, NULL, PRIO_CAN_RX, NULL);

    // RS485 接收 (CPU1 或 CPU0)
    xTaskCreate(rs485_rx_task, "RS485_RX", STACK_RS485_RX, NULL, PRIO_RS485_RX, NULL);

    // USB 发送
    xTaskCreate(usb_tx_task, "USB_TX", STACK_USB_TX, NULL, PRIO_USB_TX, NULL);

    // 状态监测 (最低优先级)
    xTaskCreate(monitor_task, "MONITOR", STACK_MONITOR, NULL, PRIO_MONITOR, NULL);
}
```

## 5. 应用层实现

### 5.1 CAN 汇聚任务

```c
void can_rx_task(void *arg)
{
    can_message_t msg;

    while (1) {
        // 从统一 CAN 队列接收 (所有 8 路 CAN 中断送数据到此队列)
        if (xQueueReceive(can_rx_queue, &msg, pdMS_TO_TICKS(10))) {
            // 封装上行协议帧
            usb_frame_t frame;
            frame.header = 0x55AA;
            frame.src_type = 0x01;          // CAN
            frame.src_ch = msg.channel;     // 通道号 0~7
            frame.timestamp = msg.timestamp;
            frame.length = sizeof(can_message_t);
            memcpy(frame.payload, &msg, sizeof(can_message_t));
            frame.crc32 = calculate_crc(frame);

            // 优先级调度: 故障帧/急停帧优先入队
            if (msg.flags & FLAG_ERROR) {
                xQueueSendToFront(usb_tx_queue, &frame, 0);  // 插队
            } else {
                xQueueSend(usb_tx_queue, &frame, 0);
            }
        }
    }
}
```

### 5.2 USB 发送任务

```c
void usb_tx_task(void *arg)
{
    usb_frame_t frame;

    while (1) {
        if (xQueueReceive(usb_tx_queue, &frame, pdMS_TO_TICKS(100))) {
            if (!usb_send_data((uint8_t *)&frame, sizeof(frame))) {
                // USB 未连接或发送失败, 缓存处理
                cache_push(&tx_cache, &frame);
            }
        }

        // 检查并发送缓存中的数据 (断线重连)
        if (tud_ready() && cache_count(&tx_cache) > 0) {
            cache_pop_and_send(&tx_cache, usb_send_data);
            // 或使用优先级: 先发实时数据, 再发缓存数据
            xQueueSend(usb_tx_queue, &frame, 0); // 当有实时数据时插队
        }
    }
}
```

### 5.3 上位机通信协议

```c
// === 上行帧 (信号板 → PC) ===
typedef struct __attribute__((packed)) {
    uint16_t sync;          // 帧头: 0x55AA
    uint8_t  src_type;      // 源类型: 0x01=CAN, 0x02=RS485, 0x03=UART, 0x04=ETH
    uint8_t  src_ch;        // 通道号: 0~7
    uint32_t timestamp;     // 时间戳 (μs)
    uint16_t length;        // 负载长度
    uint8_t  payload[256];  // 负载数据
    uint8_t  status;        // 状态字
    uint32_t crc32;         // CRC32 校验
    uint16_t tail;          // 帧尾: 0x0D0A
} usb_frame_t;

// === 下行帧 (PC → 信号板, 配置下发) ===
typedef struct __attribute__((packed)) {
    uint16_t sync;          // 帧头: 0x55BB
    uint8_t  cmd_type;      // 命令类型: 0x01=CAN配置, 0x02=RS485配置
    uint8_t  target_ch;     // 目标通道
    uint8_t  params[128];   // 参数
    uint32_t crc32;
    uint16_t tail;          // 帧尾: 0x0D0A
} host_command_t;

// 命令类型定义
#define CMD_CAN_BITRATE     0x01    // 设置 CAN 波特率
#define CMD_CAN_FILTER      0x02    // 设置 CAN ID 过滤器
#define CMD_RS485_CONFIG    0x03    // 设置 RS485 参数
#define CMD_UART_CONFIG     0x04    // 设置 UART 参数
#define CMD_DEVICE_RESET    0x10    // 设备复位
#define CMD_FIRMWARE_UPDATE 0x11    // 固件升级

// CRC32 计算
uint32_t calculate_crc(void *data, uint32_t len)
{
    // 使用硬件 CRC 或软件查表法
    return crc32_hardware(data, len);
}
```

### 5.4 RS485 驱动

```c
#include "hpm_uart_drv.h"

// RS485 通道配置
typedef struct {
    UART_Type *base;
    GPIO_Type *gpio;
    uint8_t de_pin;     // DE/RE 控制引脚
} rs485_ch_config_t;

rs485_ch_config_t rs485_chs[8] = {
    { HPM_UART1, GPIO, 10 },  // 通道0: UART1, PA10
    { HPM_UART2, GPIO, 11 },  // 通道1: UART2, PA11
    // ... 通道2~7
};

// RS485 发送
void rs485_send(uint8_t ch, uint8_t *data, uint32_t len)
{
    // 切换为发送模式
    gpio_write_pin(rs485_chs[ch].gpio, rs485_chs[ch].de_pin, 1);

    // 发送数据 (DMA 或轮询)
    uart_write_bytes(rs485_chs[ch].base, data, len);

    // 等待发送完成
    while (!uart_check_status(rs485_chs[ch].base, uart_stat_tx_complete));

    // 切回接收模式
    gpio_write_pin(rs485_chs[ch].gpio, rs485_chs[ch].de_pin, 0);
}

// RS485 接收 (DMA + 环形缓冲区)
void rs485_rx_dma_init(uint8_t ch)
{
    uint8_t *rx_buf = pvPortMalloc(RX_BUF_SIZE);
    uart_dma_rx_start(rs485_chs[ch].base, rx_buf, RX_BUF_SIZE);
}
```

```c
// 状态监测任务
void monitor_task(void *arg)
{
    uint32_t last_can_count = 0;
    uint32_t last_rs485_count = 0;

    while (1) {
        uint32_t now_can = can_total_rx_count;
        uint32_t now_rs485 = rs485_total_rx_count;

        // 检查是否有数据流动
        if (now_can == last_can_count && now_rs485 == last_rs485_count) {
            idle_counter++;
            if (idle_counter > IDLE_TIMEOUT) {
                // 全部总线静默 -> 可能异常
                log_warn("All buses idle for %d seconds", idle_counter);
            }
        } else {
            idle_counter = 0;
        }

        // 检查 CAN 总线错误
        for (int i = 0; i < 8; i++) {
            mcan_error_count_t err;
            mcan_get_error_count(HPM_MCAN0 + i, &err);
            if (err.tx_err_count > 127 || err.rx_err_count > 127) {
                log_err("CAN%d bus-off! Reset...", i);
                mcan_leave_normal_mode(HPM_MCAN0 + i);
                vTaskDelay(pdMS_TO_TICKS(10));
                mcan_enter_normal_mode(HPM_MCAN0 + i);
            }
        }

        // 喂狗
        wdg_feed(HPM_WDG0);

        last_can_count = now_can;
        last_rs485_count = now_rs485;
        vTaskDelay(pdMS_TO_TICKS(1000));  // 每秒检查一次
    }
}
```

```c
void mcan_loopback_test(MCAN_Type *base, uint8_t ch)
{
    mcan_config_t cfg;
    mcan_get_default_config(&cfg);
    cfg.mode = mcan_mode_loopback_internal;
    mcan_init(base, &cfg);

    // 发送测试帧
    can_message_t tx_msg = {
        .channel = ch,
        .id = 0x123,
        .dlc = 8,
        .data = {0,1,2,3,4,5,6,7},
    };

    mcan_send(base, &tx_msg);

    // 接收 (回环模式会自己收到)
    mcan_rx_message_t rx_msg;
    if (mcan_read_rx_fifo(base, &rx_msg) == status_success) {
        printf("CAN%d Loopback OK: ID=0x%03x, DLC=%d\n",
               ch, rx_msg.id, rx_msg.dlc);
    }
}
```

### 8.3 常见问题

| 问题 | 可能原因 | 解决方案 |
|------|----------|----------|
| JTAG 连不上 | TRST 未上拉 | 确认 PA08 上拉 10kΩ 到 3.3V |
| UART0 无输出 | 引脚被 CAN0 占用 | 确认 MCAN0 未使能 |
| CAN 总线错误 | 终端电阻未匹配 | 检查 120Ω 终端电阻配置 |
| USB 枚举失败 | 差分阻抗不匹配 | 检查 90Ω 阻抗控制 |
| 启动失败 | BOOT_MODE 配置错误 | 确认 PA02/PA03 电平 |
